## Usage

<p>
	With the <strong>{meta.title}</strong> action, a download will occur with a given Blob object as a
	file with the given filename
</p>

```tsx
<script>
	import { download } from '@svelteuidev/actions';
</script>

<div>
    <button
        use:download={{ blob: new Blob([]), filename: "test.txt" }}
        on:usedownload={() => console.log('File Downloaded')}
    >
        Download
    </button>
</div>
```

## Params

| Param    | Description                                       |
| -------- | ------------------------------------------------- |
| blob     | The Blob object representing immutable, raw data. |
| filename | The name of the file to be downloaded.            |

## Events

<p>
	The <strong>{meta.title}</strong> action dispatches custom events. Each custom event takes a
	callback function just like other events `e.g. on:click=
	{() => {}}`
</p>

```ts
on:usedownload?: (callback: (any) => unknown) => void;
on:usedownload-error?: (callback: (any) => unknown) => void;
```

## Definition

```ts
export function download(
	node: HTMLElement,
	params: { blob: Blob; filename: string }
): ReturnType<Action>;
```
